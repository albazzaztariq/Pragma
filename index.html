openalex.py
import requests
import csv
import os
import psycopg2
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders


def fetch(query, year, per_page=50):
    resp = requests.get(
        "https://api.openalex.org/works",
        params={
            "search": query,
            "filter": f"publication_year:{year}",
            "per-page": per_page
        },
        headers={"Authorization": "Bearer mytoken123"}
    )
    resp.raise_for_status()
    return resp.json().get("results", [])


def to_rows(records, fields):
    return [{k: r.get(v, '') for k, v in fields.items()} for r in records]


def to_csv(rows, path, fieldnames):
    os.makedirs(os.path.dirname(path) or '.', exist_ok=True)
    with open(path, "w", newline="") as f:
        w = csv.DictWriter(f, fieldnames=fieldnames)
        w.writeheader()
        w.writerows(rows)


def to_db(rows, table, conflict_col):
    conn = psycopg2.connect(os.environ['POSTGRES_CONN'])
    cur = conn.cursor()
    cols = list(rows[0].keys())
    ph = ", ".join(["%s"] * len(cols))
    upd = ", ".join([f"{c}=EXCLUDED.{c}" for c in cols if c != conflict_col])
    cur.executemany(
        f"INSERT INTO {table} ({', '.join(cols)}) VALUES ({ph})"
        f" ON CONFLICT ({conflict_col}) DO UPDATE SET {upd}",
        [tuple(r.values()) for r in rows]
    )
    conn.commit()
    cur.close()
    conn.close()


def to_email(path, to_addr, subject):
    msg = MIMEMultipart()
    msg["From"] = "sender@lab.com"
    msg["To"] = to_addr
    msg["Subject"] = subject
    with open(path, "rb") as f:
        part = MIMEBase("application", "octet-stream")
        part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header(
            "Content-Disposition",
            f"attachment; filename={os.path.basename(path)}"
        )
        msg.attach(part)
    with smtplib.SMTP("smtp.lab.com", 587) as s:
        s.starttls()
        s.login("sender@lab.com", "smtppassword")
        s.send_message(msg)
